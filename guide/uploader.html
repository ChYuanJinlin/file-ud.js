import { useEffect, useRef, useState } from "react";
import { FileUD, type Uploader } from "@file-ud.js/core";

export function LogoUploader() {
  const uploaderRef = useRef<Uploader | null>(null);
  const [percent, setPercent] = useState(0);
  const [loading, setLoading] = useState(false);
  const [fileName, setFileName] = useState("");
  const [previewUrl, setPreviewUrl] = useState("");
  const [serverUrl, setServerUrl] = useState("");

  useEffect(() => {
    const uploader = FileUD.createUploader("tagLogoUploader", {
      action: "/api/upload-logo",
      multiple: false,
      accept: ["image/*"],
    });

    uploader.onUpdate = () => {
      setLoading(Boolean(uploader.loading));
      setPercent(uploader.totalPercent || 0);
    };

    uploader.onSuccess = (response) => {
      setServerUrl(response.url);
    };

    uploaderRef.current = uploader;

    return () => {
      FileUD.destroyUploaders("tagLogoUploader");
      uploaderRef.current = null;
    };
  }, []);

  return (
    <>
      <button
        type="button"
        onClick={() =>
          uploaderRef.current?.open((file) => {
            setFileName(file.fileName);
            setPreviewUrl(file.url);
          })
        }
      >
        {loading ? "涓婁紶涓?quot; : "涓婁紶 Logo"}锛歿percent}%
      </button>
      {fileName ? <span>{fileName}</span> : null}
      {previewUrl ? <img src={previewUrl} alt="Logo 棰勮" style={{ width: 80, height: 80 }} /> : null}
      {serverUrl ? <input type="hidden" name="logoUrl" value={serverUrl} /> : null}
    </>
  );
}
